                                 80  * idx / (steps - 1) - 50; if idx >= steps / 4
                                 160 * idx / (steps - 1) - 70; if idx >= steps / 16
                                 480 * idx / (steps - 1) - 90; if idx > 0
                                 -144;                         if idx = 0
